\section*{Заняття 9: Диференціальні рівняння 1-го порядку, не розв'язані відносно похідної. Метод параметризації}

\subsection*{Аудиторні задачі}

Знайти загальні розв'язки і загальні інтеграли рівнянь

\begin{problem}
	$x^3 \cdot (y')^2 + x^2 \cdot y \cdot y' + a = 0$.
\end{problem}

\begin{problem}
	$x \cdot (y')^2 - 2 y' - y = 0$.
\end{problem}

\begin{problem}
	$y = 2 x \cdot y' + \sqrt{1 + (y')^2}$.
\end{problem}

\begin{problem}
	$x \cdot \sqrt{1 + (y')^2} - y' = 0$.
\end{problem}

\begin{problem}
	$x = y' \cdot \sin (y')$.
\end{problem}

\begin{problem}
	$3 (y')^5 - y \cdot y' + 1 = 0$.
\end{problem}

\begin{problem}
	$x^3 + (y')^3 - 3 x \cdot y' = 0$.
\end{problem}

\begin{problem}
	$(y')^3 - 1 = 0$.
\end{problem}

\begin{problem}
	$x \cdot (2 + (y')^2) = 1$.
\end{problem}

\begin{problem}
	$y = y' \cdot \ln (y')$.
\end{problem}
