_–ë–µ–∑ –¥–æ–≤–µ–¥–µ–Ω–Ω—è._

–Ü —Ö–æ—á–∞ –º–∏ –Ω–µ –º–æ–∂–µ–º–æ –ø–µ—Ä–µ–¥–±–∞—á–∏—Ç–∏ –Ω–∞ —â–æ –∫–æ–Ω–∫—Ä–µ—Ç–Ω–æ —Å–ø–æ–∂–∏–≤–∞—á –≤–∏—Ä—ñ—à–∏—Ç—å –≤–∏—Ç—Ä–∞—Ç–∏—Ç–∏ —Å–≤—ñ–π –¥–æ—Ö—ñ–¥, –ø—Ä–æ—Ç–µ –º–∏ –º–æ–∂–µ–º–æ —Å—Ñ–æ—Ä–º—É–ª—é–≤–∞—Ç–∏ –æ—Å–Ω–æ–≤–Ω—ñ –ø—Ä–∏–ø—É—â–µ–Ω–Ω—è, —è–∫—ñ –≤–∏–∑–Ω–∞—á–∞—é—Ç—å —Ç–µ —á–∏ —ñ–Ω—à–µ —Ä—ñ—à–µ–Ω–Ω—è —Å–ø–æ–∂–∏–≤–∞—á–∞, –∞ —Ç–∞–∫–æ–∂ —É–º–æ–≤–∏, –∑–∞ —è–∫–∏—Ö –∑–∞–±–µ–∑–ø–µ—á—É—î—Ç—å—Å—è _–º–∞–∫—Å–∏–º—ñ–∑–∞—Ü—ñ—è_ –π–æ–≥–æ –∫–æ—Ä–∏—Å–Ω–æ—Å—Ç—ñ.

2. –º–æ–∂–µ –ø–æ—Ä—ñ–≤–Ω—é–≤–∞—Ç–∏ –¥–æ—Å—Ç—É–ø–Ω—ñ –π–æ–º—É –Ω–∞–±–æ—Ä–∏ —Ç–æ–≤–∞—Ä—ñ–≤;

_–ö—Ä–∏–≤–æ—é –±–∞–π–¥—É–∂–æ—Å—Ç—ñ_ –±—É–¥–µ–º–æ –Ω–∞–∑–∏–≤–∞—Ç–∏ –ª—ñ–Ω—ñ—é (—É –≤–∏—â–∏—Ö —Ä–æ–∑–º—ñ—Ä–Ω–æ—Å—Ç—è—Ö &mdash; –ø–æ–≤–µ—Ä—Ö–Ω—é —ñ –≥—ñ–ø–µ—Ä–ø–æ–≤–µ—Ä—Ö–Ω—é –≤—ñ–¥–ø–æ–≤—ñ–¥–Ω–æ) —Ä—ñ–≤–Ω–æ—ó –∫–æ—Ä–∏—Å–Ω–æ—Å—Ç—ñ, –≤—Å—ñ —Ç–æ—á–∫–∏ —è–∫–æ—ó —Ö–∞—Ä–∞–∫—Ç–µ—Ä–∏–∑—É—é—Ç—å –Ω–∞–±–æ—Ä–∏ —Ç–æ–≤–∞—Ä—ñ–≤, —â–æ –∑–∞–±–µ–∑–ø–µ—á—É—é—Ç—å —Å–ø–æ–∂–∏–≤–∞—á–µ–≤—ñ –æ–¥–∏–Ω —ñ —Ç–æ–π –∂–µ —Ä—ñ–≤–µ–Ω—å –∫–æ—Ä–∏—Å–Ω–æ—Å—Ç—ñ:

–Ø–∫ –±–∞—á–∏–º–æ, –¥–ª—è —Ç–æ–≥–æ, —â–æ–± —Å–ø–æ–∂–∏—Ç–∏ –±—ñ–ª—å—à—É –∫—ñ–ª—å–∫—ñ—Å—Ç—å –æ–¥–Ω–æ–≥–æ –±–ª–∞–≥–∞ –≤ –Ω–∞–±–æ—Ä—ñ —Å–ø–æ–∂–∏–≤–∞—á –ø–æ–≤–∏–Ω–µ–Ω –≤—ñ–¥–º–æ–≤–∏—Ç–∏—Å—è –≤—ñ–¥ –ø–µ–≤–Ω–æ—ó –∫—ñ–ª—å–∫–æ—Å—Ç—ñ —ñ–Ω—à–æ–≥–æ –±–ª–∞–≥–∞.

–ö—ñ–ª—å–∫—ñ—Å—Ç—å –±–ª–∞–≥–∞ $$y$$ –≤—ñ–¥ —è–∫–æ—ó –ø–æ—Ç—Ä—ñ–±–Ω–æ –≤—ñ–¥–º–æ–≤–∏—Ç–∏—Å—è –≤ –æ–±–º—ñ–Ω –Ω–∞ –æ–¥–Ω—É –¥–æ–¥–∞—Ç–∫–æ–≤—É –æ–¥–∏–Ω–∏—Ü—é –±–ª–∞–≥–∞ $$x$$ –ø—Ä–∏ –Ω–µ–∑–º—ñ–Ω–Ω–æ–º—É –∑–∞–≥–∞–ª—å–Ω–æ–º—É —Ä—ñ–≤–Ω—ñ –∫–æ—Ä–∏—Å–Ω–æ—Å—Ç—ñ –Ω–∞–∑–∏–≤–∞—î—Ç—å—Å—è _–≥—Ä–∞–Ω–∏—á–Ω–æ—é –Ω–æ—Ä–º–æ—é –∑–∞–º—ñ—â–µ–Ω–Ω—è_ –±–ª–∞–≥:

$$
\frac{\mathrm{d} y}{\mathrm{d} x} = \displaystyle\lim\limits_{\Delta x \to 0} \left( - \frac{\Delta y}{\Delta x} \right).
$$

–í —Ü—ñ–ª–æ–º—É –∫—Ä–∏–≤–∏–º –±–∞–π–¥—É–∂–æ—Å—Ç—ñ –ø—Ä–∏—Ç–∞–º–∞–Ω–Ω—ñ —Ç–∞–∫—ñ –≤–ª–∞—Å—Ç–∏–≤–æ—Å—Ç—ñ:

- –º–∞—é—Ç—å –≤—ñ–¥'—î–º–Ω–∏–π –Ω–∞—Ö–∏–ª;

- –Ω–µ –ø–µ—Ä–µ—Ç–∏–Ω–∞—é—Ç—å—Å—è;

- –¥–∞–ª—ñ –≤—ñ–¥ –ø–æ—á–∞—Ç–∫—É –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç &mdash; –±—ñ–ª—å—à–∞ –∫–æ—Ä–∏—Å–Ω—ñ—Å—Ç—å;

- –≤–∑–¥–æ–≤–∂ (–ø—Ä–∏ $$x \to \infty$$) –∫—Ä–∏–≤–æ—ó –±–∞–π–¥—É–∂–æ—Å—Ç—ñ –≥—Ä–∞–Ω–∏—á–Ω–∞ –Ω–æ—Ä–º–∞ –∑–∞–º—ñ—â–µ–Ω–Ω—è –∑–º–µ–Ω—à—É—î—Ç—å—Å—è.

#### 4. –ù–µ–æ–∫–ª–∞—Å–∏—á–Ω–∞ –∑–∞–¥–∞—á–∞ —Å–ø–æ–∂–∏–≤–∞–Ω–Ω—è. –£–º–æ–≤–∏ –æ–ø—Ç–∏–º–∞–ª—å–Ω–æ—Å—Ç—ñ.

–Ø–∫—â–æ —Ñ—É–Ω–∫—Ü—ñ—è –∫–æ—Ä–∏—Å–Ω–æ—Å—Ç—ñ —î –¥–≤—ñ—á—ñ –Ω–µ–ø–µ—Ä–µ—Ä–≤–Ω–æ-–¥–∏—Ñ–µ—Ä–µ–Ω—Ü—ñ–π–æ–≤–Ω–æ—é —Ç–∞ —Å—Ç—Ä–æ–≥–æ –æ–ø—É–∫–ª–æ—é, –∞ –±—é–¥–∂–µ—Ç–Ω–µ –æ–±–º–µ–∂–µ–Ω–Ω—è –º–∞—î –≤–∏–≥–ª—è–¥ $$\langle p, x \rangle \le I$$, —Ç–æ —Ä–∞—Ü—ñ–æ–Ω–∞–ª—å–Ω–∞ –ø–æ–≤–µ–¥—ñ–Ω–∫–∏ —Å–ø–æ–∂–∏–≤–∞—á–∞ –≤–∏–∑–Ω–∞—á–∞—î—Ç—å—Å—è —Ç–∞–∫–æ—é –∑–∞–¥–∞—á–µ—é

$$
u(x) \xrightarrow[\substack{\langle p, x\rangle \le I \\ x \in \mathbb{R}_+^n}]{} \max.
$$

–û—Å–∫—ñ–ª—å–∫–∏ –¥–æ–ø—É—Å—Ç–∏–º–∞ –º–Ω–æ–∂–∏–Ω–∞ —î –∫–æ–º–ø–∞–∫—Ç–Ω–æ—é, –æ–ø—É–∫–ª–æ—é, —ñ –∑–∞–º–∫–Ω–µ–Ω–æ—é, —Ç–æ –∑–∞–¥–∞—á–∞ –º–∞—î —î–¥–∏–Ω–∏–π —Ä–æ–∑–≤'—è–∑–æ–∫ $$x^\star$$. _–£ —Ü—ñ–π –º–æ–¥–µ–ª—ñ $$x^\star = x^\star(p, I)$$._

–†–æ–∑–≥–ª—è–Ω–µ–º–æ —Ñ—É–Ω–∫—Ü—ñ—é –õ–∞–≥—Ä–∞–Ω–∂–∞ —Ü—ñ—î—ó –∑–∞–¥–∞—á—ñ:

$$
L(x, \lambda) = u(x) + \lambda (I - \langle p, x \rangle).
$$

–Ø–∫ –≤—ñ–¥–æ–º–æ, –Ω–µ–æ–±—Ö—ñ–¥–Ω–∏–º–∏ —É–º–æ–≤–∞–º–∏ –æ–ø—Ç–∏–º–∞–ª—å–Ω–æ—Å—Ç—ñ —î –Ω–∞—Å—Ç—É–ø–Ω—ñ:

$$
\left\{
\begin{aligned}
& \frac{\partial L(x^\star, \lambda^\star)}{\partial x} = Mu(x^\star) - \lambda^\star p \le 0, \\
& \frac{\partial L(x^\star, \lambda^\star)}{\partial \lambda} = I - \langle p, x \rangle \ge 0, \\
& \left\langle x^\star, \frac{\partial L(x^\star, \lambda^\star)}{\partial x} \right\rangle = \left\langle x^\star, Mu(x^\star) - \lambda^\star p \right\rangle = 0, \\
& \lambda^\star \cdot \frac{\partial L(x^\star, \lambda^\star)}{\partial \lambda} = \lambda^\star \cdot (I - \langle p, x \rangle) = 0, \\
& x^\star, \lambda^\star \ge 0.
\end{aligned}
\right.
$$

–û–ø—Ç–∏–º–∞–ª—å–Ω–∏–π (—à—Ç—Ä–∞—Ñ–Ω–∏–π) –º–Ω–æ–∂–Ω–∏–∫ $$\lambda^\star$$ –º–æ–∂–Ω–∞ —ñ–Ω—Ç–µ—Ä–ø—Ä–µ—Ç—É–≤–∞—Ç–∏ —è–∫ –≥—Ä–∞–Ω–∏—á–Ω—É –∫–æ—Ä–∏—Å–Ω—ñ—Å—Ç—å –¥–æ–¥–∞—Ç–∫–æ–≤–æ–≥–æ –¥–æ—Ö–æ–¥—É:

$$
\lambda^\star = \frac{\partial u(x^\star(p, I))}{\partial I}.
$$

–ú–æ–∂–Ω–∞ —Ç–∞–∫–æ–∂ —ñ–Ω—Ç–µ—Ä–ø—Ä–µ—Ç—É–≤–∞—Ç–∏ –∑–∞–¥–∞—á—É –≥–µ–æ–º–µ—Ç—Ä–∏—á–Ω–æ: —Ä–æ–∑–≤'—è–∑–æ–∫ –ª–µ–∂–∏—Ç—å –Ω–∞ _–±—é–¥–∂–µ—Ç–Ω—ñ–π –ª—ñ–Ω—ñ—ó_ (–≥–µ–æ–º–µ—Ç—Ä—á–∏–Ω–µ –º—ñ—Å—Ü–µ —Ç–æ—á–æ–∫ —è–∫—ñ –≤—ñ–¥–ø–æ–≤—ñ–¥–∞—é—Ç—å –Ω–∞–±–æ—Ä–∞–º —Ç–æ–≤–∞—Ä—ñ–≤ –∑–∞ —è–∫–∏—Ö —Å–ø–æ–∂–∏–≤–∞—á –æ–ø–≤–Ω—ñ—Å—Ç—é –≤–∏—Ç—Ä–∞—á–∞—î —Å–≤—ñ–π –¥–æ—Ö—ñ–¥).

#### 5. –ó–∞–¥–∞—á–∞ —Å–ø–æ–∂–∏–≤–∞–Ω–Ω—è –∑–∞ –•—ñ–∫—Å–æ–º. –£–º–æ–≤–∏ –æ–ø—Ç–∏–º–∞–ª—å–Ω–æ—Å—Ç—ñ.

–†–æ–∑–≥–ª—è–Ω–µ–º–æ –∑–∞–¥–∞—á—É —Å–ø–æ–∂–∏–≤–∞–Ω–Ω—è –∑–∞ –•—ñ–∫—Å–æ–º:

$$
\displaystyle\sum\limits_{i = 1}^n p_i x_i \xrightarrow[\substack{u(x) \ge u_0 \\ x \in \mathbb{R}_+^n}]{} \min,
$$

—Ç–æ–±—Ç–æ –≤—ñ–¥–±—É–≤—ñ—î—Ç—å—Å—è –º—ñ–Ω—ñ–º—ñ–∑–∞—Ü—ñ—è –≤—Ç—Ä–∞—Ç –∑–∞ —É–º–æ–≤–∏ –¥–æ—Å—è–≥–Ω–µ–Ω–Ω—è –º—ñ–Ω—ñ–º–∞–ª—å–Ω–æ–≥–æ —Ä—ñ–≤–Ω—è –∫–æ—Ä–∏—Å–Ω–æ—Å—Ç—ñ. _–£ —Ü—ñ–π –º–æ–¥–µ–ª—ñ $$x^\star = x^\star(p, u_0)$$._

–î–ª—è —Ü—ñ—î—ó –∑–∞–¥–∞—á—ñ —Ç–∞–∫–æ–∂ –±—É–¥—É—î—Ç—å—Å—è (–≤–∂–µ —ñ–Ω—à–∞) —Ñ—É–Ω–∫—Ü—ñ—è –õ–∞–≥—Ä–∞–Ω–∂–∞:

$$
L(x, \lambda) = - \displaystyle\sum\limits_{i = 1}^n p_i x_i + \lambda (u(x) - u_0),
$$

–¥–ª—è —è–∫–æ—ó –º–∞—é—Ç—å –≤–∏–∫–æ–Ω—É–≤–∞—Ç–∏—Å—è –Ω–∞—Å—Ç—É–ø–Ω—ñ —É–º–æ–≤–∏ –æ–ø—Ç–∏–º–∞–ª—å–Ω–æ—Å—Ç—ñ:

$$
\left\{
\begin{aligned}
& \frac{\partial L(x^\star, \lambda^\star)}{\partial x} = - p + \lambda Mu = 0, \\
& \frac{\partial L(x^\star, \lambda^\star)}{\partial \lambda} = u(x) - u_0 = 0.
\end{aligned}
\right.
$$

–û—Ç—Ä–∏–º–∞–Ω–∏–π –≤ —Ä–µ–∑—É–ª—å—Ç–∞—Ç—ñ –≤–µ–∫—Ç–æ—Ä $$x^\star$$ –Ω–∞–∑–∏–≤–∞—î—Ç—å—Å—è —Ç–∞–∫–æ–∂ _—Ö—ñ–∫—Å—ñ–∫–∞–Ω—Å—å–∫–∏–º –ø–æ–ø–∏—Ç–æ–º._

### –¢–µ–æ—Ä—ñ—è –≤–∏—Ä–æ–±–Ω–∏—Ü—Ç–≤–∞

#### 1. –ï–ª–∞—Å—Ç–∏—á–Ω—ñ—Å—Ç—å –≤–∏–ø—É—Å–∫—É. –ï–ª–∞—Å—Ç–∏—á–Ω—ñ—Å—Ç—å –∑–∞–º—ñ—â–µ–Ω–Ω—è. –ú–∞—Å—à—Ç–∞–± –≤–∏—Ä–æ–±–Ω–∏—Ü—Ç–≤–∞.

–õ–æ–≥—ñ—á–Ω–∏–º —î –ø–∏—Ç–∞–Ω–Ω—è –∑–∞–ª–µ–∂–Ω–æ—Å—Ç—ñ –æ–±—Å—è–≥—ñ–≤ –≤–∏—Ä–æ–±–Ω–∏—Ü—Ç–≤–∞ $$Q$$ –≤—ñ–¥ –º–∞—Å—à—Ç–∞–±—É –≤–∏—Ä–æ–±–Ω–∏—Ü—Ç–≤–∞ $$m$$.

_–ï–ª–∞—Å—Ç–∏—á–Ω—ñ—Å—Ç—å –≤–∏–ø—É—Å–∫—É –ø–æ –º–∞—Å—à—Ç–∞–±—É_ –≤–∏–∑–Ω–∞—á–∞—î—Ç—å—Å—è —è–∫

$$
E_{Q, m} = \frac{\frac{\Delta Q}{Q} \cdot 100\%}{\frac{\Delta m}{m} \cdot 100\%} = \frac{\Delta Q}{\delta m} \cdot \frac{m}{Q}.
$$

–ó –º–∞—Ç–µ–º–∞—Ç–∏—á–Ω–æ–≥–æ –ø–æ–≥–ª—è–¥—É –∫–æ–µ—Ñ—ñ—Ü—ñ—î–Ω—Ç –µ–ª–∞—Å—Ç–∏—á–Ω–æ—Å—Ç—ñ –º–∞—Å—à—Ç–∞–±—É —Ö–∞—Ä–∞–∫—Ç–µ—Ä–∏–∑—É—î —Å—Ç—É–ø—ñ–Ω—å –æ–¥–Ω–æ—Ä—ñ–¥–Ω–æ—Å—Ç—ñ –≤–∏—Ä–æ–±–Ω–∏—á–æ—ó —Ñ—É–Ω–∫—Ü—ñ—ó: 

$$
Q(m K, m L) = Q(K, L) \cdot m^{E_{Q, m}} = Q_0 \cdot m^{E_{Q, m}}.
$$

–£—è–≤–ª—è—Ç–∏ —Å–æ–±—ñ —Ü–µ –º–æ–∂–Ω–∞ —Ç–∞–∫:

–ó–∞ –Ω–µ—Å–∫—ñ–Ω—á–µ–Ω–Ω–æ –º–∞–ª–æ—ó –∑–º—ñ–Ω–∏ —Ä—ñ–≤–Ω—è –∑–∞—Å—Ç–æ—Å–æ–≤—É–≤–∞–Ω–∏—Ö —Ñ–∞–∫—Ç–æ—Ä—ñ–≤ –∑–º—ñ–Ω—É –≤–∏–ø—É—Å–∫—É –¥–ª—è –æ–¥–Ω–æ—Ä—ñ–¥–Ω–æ—ó —Ñ—É–Ω–∫—Ü—ñ—ó —Ç–∏–ø—É $$Q = F (K, L)$$ –º–æ–∂–Ω–∞ –ø–æ–¥–∞—Ç–∏ —á–µ—Ä–µ–∑ –ø–æ–≤–Ω–∏–π –¥–∏—Ñ–µ—Ä–µ–Ω—Ü—ñ–∞–ª:

$$
\mathrm{d} Q = \frac{\partial F}{\partial L} \cdot \mathrm{d} L + \frac{\partial F}{\partial K} \cdot \mathrm{d} K.
$$

–¢—É—Ç –∫–æ–∂–Ω–∏–π –∑ –¥–æ–¥–∞–Ω–∫—ñ–≤ —Ö–∞—Ä–∞–∫—Ç–µ—Ä–∏–∑—É—î –∑–º—ñ–Ω—É –≤–∏–ø—É—Å–∫—É –∑–∞ —Ä–∞—Ö—É–Ω–æ–∫ –∑–º—ñ–Ω–∏ –≤—ñ–¥–ø–æ–≤—ñ–¥–Ω–æ–≥–æ —Ñ–∞–∫—Ç–æ—Ä–∞.

–ó–∞ –ø—Ä–æ–ø–æ—Ä—Ü—ñ–π–Ω–æ—ó –≤–∞—Ä—ñ–∞—Ü—ñ—ó —Ñ–∞–∫—Ç–æ—Ä—ñ–≤ –≤–∏–∫–æ–Ω—É—é—Ç—å—Å—è —É–º–æ–≤–∏

$$
\mathrm{d} L = \frac{\mathrm{d} m}{m} \cdot L, \quad \mathrm{d} K = \frac{\mathrm{d} m}{m} \cdot K.
$$

–ü—ñ–¥—Å—Ç–∞–≤–ª—è—é—á–∏ —Ü–µ–π –≤–∏—Ä–∞–∑ —É –ø–æ–ø–µ—Ä–µ–¥–Ω—ñ–π –∑–Ω–∞—Ö–æ–¥–∏–º–æ

$$
\mathrm{d} Q = \frac{\partial F}{\partial L} \cdot \frac{\mathrm{d} m}{m} \cdot L + \frac{\partial F}{\partial K} \frac{\mathrm{d} m}{m} \cdot K,
$$

–∑–≤—ñ–¥–∫–∏ –∑–Ω–∞—Ö–æ–¥–∏–º–æ

$$
\frac{\mathrm{d} Q}{\mathrm{d} m} \cdot \frac{m}{Q} = \frac{\partial F}{\partial L} \cdot \frac{L}{Q} + \frac{\partial F}{\partial K} \cdot \frac{K}{Q},
$$

—Ç–æ–±—Ç–æ –µ–ª–∞—Å—Ç–∏—á–Ω—ñ—Å—Ç—å –≤–∏–ø—É—Å–∫—É –∑–∞ –º–∞—Å—à—Ç–∞–±–æ–º –¥–æ—Ä—ñ–≤–Ω—é—î —Å—É–º—ñ –µ–ª–∞—Å—Ç–∏—á–Ω–æ—Å—Ç–µ–π –≤–∏–ø—É—Å–∫—É –∑–∞ –≤–∏–∫–æ—Ä–∏—Å—Ç–æ–≤—É–≤–∞–Ω–∏–º–∏ —Ñ–∞–∫—Ç–æ—Ä–∞–º–∏: 

$$
E_{Q, m} = E_{Q, L} + E_{Q, K}.
$$

–¶–µ–π —Ñ–∞–∫—Ç –Ω–∞–∑–∏–≤–∞—î—Ç—Å—å—è —Ç–µ–æ—Ä–µ–º–æ—é –í—ñ–∫—Å–µ–ª–ª—è-–î–∂–æ–Ω—Å–æ–Ω–∞.

–©–µ –æ–¥–Ω–∏–º –≤–∞–∂–ª–∏–≤–∏–º –ø–æ–∫–∞–∑–Ω–∏–∫–æ–º, —è–∫–∏–π —Ö–∞—Ä–∞–∫—Ç–µ—Ä–∏–∑—É—î —ñ–∑–æ–∫–≤–∞–Ω—Ç–Ω—É –≤–∞—Ä—ñ–∞—Ü—ñ—é, —î –µ–ª–∞—Å—Ç–∏—á–Ω—ñ—Å—Ç—å –∑–∞–º—ñ—â–µ–Ω–Ω—è —Ñ–∞–∫—Ç–æ—Ä—ñ–≤. –¶–µ –ø–æ–Ω—è—Ç—Ç—è –≤–≤–µ–¥–µ–Ω–µ –≤ –µ–∫–æ–Ω–æ–º—ñ—á–Ω—É —Ç–µ–æ—Ä—ñ—é –•—ñ–∫—Å–æ–º.

–ï–ª–∞—Å—Ç–∏—á–Ω—ñ—Å—Ç—å –∑–∞–º—ñ—â–µ–Ω–Ω—è –ø–æ–∫–∞–∑—É—î, –Ω–∞ —Å–∫—ñ–ª—å–∫–∏ –≤—ñ–¥—Å–æ—Ç–∫—ñ–≤ –∑–º—ñ–Ω–∏—Ç—å—Å—è —Å–ø—ñ–≤–≤—ñ–¥–Ω–æ—à–µ–Ω–Ω—è –≤–∏—Ç—Ä–∞—Ç –¥–≤–æ—Ö —Ñ–∞–∫—Ç–æ—Ä—ñ–≤, —è–∫—â–æ –≥—Ä–∞–Ω–∏—á–Ω–∞ –Ω–æ—Ä–º–∞ —ó—Ö–Ω—å–æ–≥–æ –∑–∞–º—ñ—â–µ–Ω–Ω—è –∑–º—ñ–Ω–∏—Ç—å—Å—è –Ω–∞ $$1\%$$ (–∑–∞ —ñ–Ω—à–∏—Ö —Ä—ñ–≤–Ω–∏—Ö —É–º–æ–≤, —Ç–æ–±—Ç–æ –∑–∞ —Å—Ç–∞–ª–æ—Å—Ç—ñ –≤–∏—Ç—Ä–∞—Ç —É—Å—ñ—Ö —ñ–Ω—à–∏—Ö —Ñ–∞–∫—Ç–æ—Ä—ñ–≤ —ñ –Ω–µ–∑–º—ñ–Ω–Ω–æ–≥–æ –≤–∏–ø—É—Å–∫—É).

–ö–æ–µ—Ñ—ñ—Ü—ñ—î–Ω—Ç –µ–ª–∞—Å—Ç–∏—á–Ω–æ—Å—Ç—ñ –∑–∞–º—ñ—â–µ–Ω–Ω—è –ø—Ä–∞—Ü–µ—é –∫–∞–ø—ñ—Ç–∞–ª—É, –Ω–∞–ø—Ä–∏–∫–ª–∞–¥, –º–æ–∂–µ –±—É—Ç–∏ –≤–∏–∑–Ω–∞—á–µ–Ω–∏–π —Ç–∞–∫:

$$
E_{K, L} = \frac{\Delta (K / L) / (K / L) \cdot 100\%}{\Delta (\text{MP}_L / \text{MP}_K) \cdot (\text{MP}_L / \text{MP}_K) \cdot 100\%}.
$$

–£ —Ç–∞–∫–æ–º—É —Ä–∞–∑—ñ –º–æ–∂–Ω–∞ —Ç–∞–∫ –≤–∏–∑–Ω–∞—á–∏—Ç–∏ –∫–æ–µ—Ñ—ñ—Ü—ñ—î–Ω—Ç –µ–ª–∞—Å—Ç–∏—á–Ω–æ—Å—Ç—ñ –∑–∞–º—ñ—â–µ–Ω–Ω—è –ø—Ä–∞—Ü–µ—é –∫–∞–ø—ñ—Ç–∞–ª—É: $$E_{L, K}$$ –ø–æ–∫–∞–∑—É—î, –Ω–∞ —Å–∫—ñ–ª—å–∫–∏ –≤—ñ–¥—Å–æ—Ç–∫—ñ–≤ –∑–º—ñ–Ω–∏—Ç—å—Å—è —Å–ø—ñ–≤–≤—ñ–¥–Ω–æ—à–µ–Ω–Ω—è –≤–∏—Ç—Ä–∞—Ç –∫–∞–ø—ñ—Ç–∞–ª—É —ñ –ø—Ä–∞—Ü—ñ, —è–∫—â–æ —Å–ø—ñ–≤–≤—ñ–¥–Ω–æ—à–µ–Ω–Ω—è —ó—Ö–Ω—ñ—Ö –≥—Ä–∞–Ω–∏—á–Ω–∏—Ö –ø—Ä–æ–¥—É–∫—Ç—ñ–≤ –∑–º—ñ–Ω–∏—Ç—å—Å—è –Ω–∞ $$1\%$$ (–∑–∞ —ñ–Ω—à–∏—Ö —Ä—ñ–≤–Ω–∏—Ö —É–º–æ–≤).

#### 2. –î–æ—Å–∫–æ–Ω–∞–ª–∞ –∫–æ–Ω–∫—É—Ä–µ–Ω—Ü—ñ—è. –ü–æ–ø–∏—Ç, –≤–∏—Ä—É—á–∫–∞ —Ç–∞ –ø—Ä–∏–±—É—Ç–æ–∫ –∫–æ–Ω–∫—É—Ä–µ–Ω—Ç–Ω–æ—ó —Ñ—ñ—Ä–º–∏.

–†–æ–∑–≥–ª—è–Ω–µ–º–æ —Ä—ñ–≤–Ω–æ–≤–∞–∂–Ω—É —Ü—ñ–Ω–∞ —Ç–∞ —Ä—ñ–≤–Ω–æ–≤–∞–∂–Ω–∏–π –æ–±—Å—è–≥ –ø—Ä–æ–¥—É–∫—Ü—ñ—ó, —â–æ –≤—Å—Ç–∞–Ω–æ–≤–∏–ª–∏—Å—è –Ω–∞ –ø–æ–≤–Ω—ñ—Å—Ç—é –∫–æ–Ω–∫—É—Ä–µ–Ω—Ç–Ω–æ–º—É —Ä–∏–Ω–∫—É, —è–∫–∏–π –ø–µ—Ä–µ–±—É–≤–∞—î —É —Å—Ç–∞–Ω—ñ —Ä—ñ–≤–Ω–æ–≤–∞–≥–∏, –≤–Ω–∞—Å–ª—ñ–¥–æ–∫ –≤–∑–∞—î–º–æ–¥—ñ—ó —Ç–∏—Å—è—á –∫–æ–Ω–∫—É—Ä–µ–Ω—Ç–Ω–∏—Ö —Ñ—ñ—Ä–º —Ç–∞ —Å–ø–æ–∂–∏–≤–∞—á—ñ–≤.

–û—Å–æ–±–ª–∏–≤—ñ—Å—Ç—é —Ç–∞–∫–æ—ó —Å–∏—Ç—É–∞—Ü—ñ—ó —î —Ç–µ, —â–æ –∂–æ–¥–Ω–∞ —Ñ—ñ—Ä–º–∞ –Ω–µ –º–æ–∂–µ –≤—ñ–¥—Ö–∏–ª–∏—Ç–∏—Å—è –≤—ñ–¥ —Ä—ñ–≤–Ω–æ–≤–∞–∂–Ω–æ—ó —Ü—ñ–Ω–∏ —ñ –∑–±—ñ–ª—å—à–∏—Ç–∏ —Å–≤—ñ–π –ø—Ä–∏–±—É—Ç–æ–∫.

–ó—Ä–æ–∑—É–º—ñ–ª–æ, —â–æ —Ç—É—Ç –º–∏ –ø—Ä–∏–ø—É—Å–∫–∞—î–º–æ, —â–æ –∂–æ–¥–µ–Ω –æ–∫—Ä–µ–º–∏–π –≤–∏—Ä–æ–±–Ω–∏–∫ –Ω–µ –º–æ–∂–µ (_—Å—É—Ç—Ç—î–≤–æ_) –≤–ø–ª–∏–Ω—É—Ç–∏ –Ω–∞ —Ä—ñ–≤–Ω–æ–≤–∞–∂–Ω—É —Ü—ñ–Ω—É –∑–º—ñ–Ω–æ—é –æ–±—Å—è–≥—É —Å–≤–æ—î—ó –ø—Ä–æ–¥—É–∫—Ü—ñ—ó, –æ—Å–∫—ñ–ª—å–∫–∏ –π–æ–≥–æ —á–∞—Å—Ç–∫–∞ —î –¥—É–∂–µ –º–∞–ª–æ—é –≤ –≥–∞–ª—É–∑–µ–≤–æ–º—É –æ–±—Å—è–≥—É –ø—Ä–æ–ø–æ–∑–∏—Ü—ñ—ó.

–ö–æ–Ω–∫—É—Ä–µ–Ω—Ç–Ω–∞ —Ñ—ñ—Ä–º–∞ –±—É–¥—å-—è–∫–∏–π –æ–±—Å—è–≥ —Å–≤–æ–≥–æ –≤–∏—Ä–æ–±–Ω–∏—Ü—Ç–≤–∞ –º–æ–∂–µ –ø—Ä–æ–¥–∞—Ç–∏ –ª–∏—à–µ –∑–∞ —Ü—ñ–Ω–æ—é —Ä–∏–Ω–∫–æ–≤–æ—ó —Ä—ñ–≤–Ω–æ–≤–∞–≥–∏, –æ—Ç–∂–µ –ø–æ–ø–∏—Ç –Ω–∞ –ø—Ä–æ–¥—É–∫—Ü—ñ—é –∫–æ–Ω–∫—É—Ä–µ–Ω—Ç–Ω–æ—ó —Ñ—ñ—Ä–º–∏ —î –∞–±—Å–æ–ª—é—Ç–Ω–æ –µ–ª–∞—Å—Ç–∏—á–Ω–∏–º, –∞ –∫—Ä–∏–≤–∞ –ø–æ–ø–∏—Ç—É —î –≥–æ—Ä–∏–∑–æ–Ω—Ç–∞–ª—å–Ω–æ—é –ª—ñ–Ω—ñ—î—é:

<img src="/eco/exams/2/img/2.png" alt="–∫—Ä–∏–≤—ñ –ø–æ–ø–∏—Ç—É —Ç–∞ –ø—Ä–æ–ø–æ–∑–∏—Ü—ñ—ó" style="width: 100%;">

–ß–µ—Ä–µ–∑ —Ü–µ —Ñ—ñ—Ä–º–∞ –Ω–∞ —Ç–∞–∫–æ–º—É —Ä–∏–Ω–∫—É –Ω–∞–∑–∏–≤–∞—î—Ç—å—Å—è _—Ü—ñ–Ω–æ–æ–¥–µ—Ä–∂—É–≤–∞—á–µ–º_, –∞–¥–∂–µ –≤–æ–Ω–∞ –Ω–µ –≤–ø–ª–∏–≤–∞—î –Ω–∞ —Ä–∏–Ω–∫–æ–≤—É —Ü—ñ–Ω—É.

–§—ñ—Ä–º–∞ –æ—Ç—Ä–∏–º—É—î _–≤–∏—Ä—É—á–∫—É_ (eng. _total revenue_):

$$
\text{TR} = p \cdot q.
$$

<!-- –¢–∞–∫–æ–∂ –≤–≤–µ–¥–µ–º–æ –ø–æ–Ω—è—Ç—Ç—è –¥–æ—Ö–æ–¥—É –Ω–∞ –æ–¥–∏–Ω–∏—Ü—é –ø—Ä–æ–¥—É–∫—Ü—ñ—ó (eng. _average revenue_) $${AR} = \text{TR} / q = p$$, —ñ –∑–º—ñ–Ω–∏ –≤–∏—Ä—É—á–∫–∏ –≤–Ω–∞—Å–ª—ñ–¥–æ–∫ –ø—Ä–æ–¥–∞–∂—É –¥–æ–¥–∞—Ç–∫–æ–≤–æ—ó –æ–¥–∏–Ω–∏—Ü—ñ –ø—Ä–æ–¥—É–∫—Ü—ñ—ó (eng. _marginal revenue_) $$\text{MR} = \mathrm{d} \text{TR} / \mathrm{d} q = p$$. -->

–û—Å–∫—ñ–ª—å–∫–∏ —Ñ—ñ—Ä–º–∞ –Ω–µ –ª–∏—à–µ –ø—Ä–æ–¥–∞—î –ø—Ä–æ–¥—É–∫—Ü—ñ—é, –∞–ª–µ –π –≤–∏—Ä–æ–±–ª—è—î —ó—ó, —Ç–æ –¥–æ—Ü—ñ–ª—å–Ω–æ —Ä–æ–∑–≥–ª—è–Ω—É—Ç–∏ —Ñ—É–Ω–∫—Ü—ñ—é —Å—É–∫—É–ø–Ω–æ—ó _–≤–∞—Ä—Ç–æ—Å—Ç—ñ_ (eng. _total cost_) –≤–∏—Ä–æ–±–Ω–∏—Ü—Ç–≤–∞:

$$
\text{TC} = \displaystyle\sum\limits_{i = 1}^m w_i x_i = \langle w, x \rangle,
$$

–¥–µ $$x_i$$ &mdash; –∫—ñ–ª—å–∫—ñ—Å—Ç—å $$i$$-–≥–æ —Ä–µ—Å—É—Ä—Å—É –≤–∏–∫–æ—Ä–∏—Å—Ç–∞–Ω–æ–≥–æ –ø—Ä–∏ –≤–∏—Ä–æ–±–Ω–∏—Ü—Ç–≤—ñ, –∞ $$w_i$$ &mdash; –≤–∞—Ä—Ç—ñ—Å—Ç—å –æ–¥–∏–Ω–∏—Ü—ñ —Ü—å–æ–≥–æ —Ä–µ—Å—É—Ä—Å—É.

–ß–µ—Ä–µ–∑ —Ç–µ, —â–æ –æ–¥–Ω–∞–∫–æ–≤—ñ –æ–±—Å—è–≥–∏ –≤–∏—Ä–æ–±–Ω–∏—Ü—Ç–≤–∞ –ø—Ä–æ–¥—É–∫—Ü—ñ—ó –º–æ–∂—É—Ç—å –∑–∞–±–µ–∑–ø–µ—á—É–≤–∞—Ç–∏—Å—è –≤–Ω–∞—Å–ª—ñ–¥–æ–∫ —Ä—ñ–∑–Ω–∏—Ö –∫–æ–º–±—ñ–Ω–∞—Ü—ñ–π –æ–±—Å—è–≥—ñ–≤ —Ä–µ—Å—É—Ä—Å—ñ–≤ –∑ —Ä—ñ–∑–Ω–æ—é –≤–∞—Ä—Ç—ñ—Å—Ç—é, —Å—Ç–∞—î –¥–æ—Ü—ñ–ª—å–Ω–∏–º —Ä–æ–∑–≥–ª—è–¥–∞—Ç–∏ _–≤–∏—Ä–æ–±–Ω–∏—á—É —Ñ—É–Ω–∫—Ü—ñ—é_: $$Q(x)$$. 

–ó—Ä–æ–∑—É–º—ñ–ª–æ, —â–æ —Ç–æ–¥—ñ _–ø—Ä–∏–±—É—Ç–æ–∫_ —Ñ—ñ—Ä–º–∏ –≤–∏–∑–Ω–∞—á–∞—î—Ç—å—Å—è —è–∫ 

$$
\pi = \text{TR} - \text{TC}.
$$

#### 3. –ú–∞–∫—Å–∏–º—ñ–∑–∞—Ü—ñ—è –ø—Ä–∏–±—É—Ç–∫—É —Ñ—ñ—Ä–º–∏ (–≥—Ä–∞—Ñ—ñ—á–Ω–∏–π —Ç–∞ –∞–Ω–∞–ª—ñ—Ç–∏—á–Ω–∏–π —Ä–æ–∑–≤'—è–∑–∫–∏).

–ó–∞–¥–∞—á–∞ –º–∞–∫—Å–∏–º—ñ–∑–∞—Ü—ñ—ó –ø—Ä–∏–±—É—Ç–∫—É –º–æ–∂–µ –±—É—Ç–∏ —Ä–æ–∑–≤'—è–∑–∞–Ω–∞ –≤ –∞–Ω–∞–ª—ñ—Ç–∏—á–Ω–æ–º—É —Ç–∞ –≥—Ä–∞—Ñ—ñ—á–Ω–æ–º—É –≤–∏–≥–ª—è–¥—ñ. –†–æ–∑–≥–ª—è–Ω–µ–º–æ —Å–ø–µ—Ä—à—É –≥—Ä–∞—Ñ—ñ—á–Ω–∏–π –≤–∏–≥–ª—è–¥:

–Ø–∫ –±–∞—á–∏–º–æ:

- –ü—Ä–∏ –º–∞–ª–∏—Ö –æ–±—Å—è–≥–∞—Ö –≤–∏—Ä–æ–±–Ω–∏—Ü—Ç–≤–∞ $$q < q_A$$ –º–∞—î–º–æ $$\text{TC} > \text{TR}$$ —ñ —Ñ—ñ—Ä–º–∞ –Ω–µ—Å–µ –∑–±–∏—Ç–∫–∏.

- –ü—Ä–∏ $$q \in (q_A, q_B)$$ –º–∞—î–º–æ $$\text{TC} < \text{TR}$$ —ñ —Ñ—ñ—Ä–º–∞ –º–∞—î —Ö–æ—á–∞ –± —è–∫–∏–π—Å—å –ø—Ä–∏–±—É—Ç–æ–∫.

- –ü—Ä–∏ –≤–µ–ª–∏–∫–∏—Ö –æ–±—Å—è–≥–∞—Ö –≤–∏—Ä–æ–±–Ω–∏—Ü—Ç–≤–∞ $$q > q_B$$ –º–∞—î–º–æ $$\text{TC} > \text{TR}$$ —ñ —Ñ—ñ—Ä–º–∞ –∑–Ω–æ–≤—É –º–∞—î –∑–±–∏—Ç–∫–∏.

–ó—Ä–æ–∑—É–º—ñ–ª–æ —Ç–∞–∫–æ–∂, —â–æ –º–∞–∫—Å–∏–º–∞–ª—å–Ω–∏–π –ø—Ä–∏–±—É—Ç–æ–∫ –¥–æ—Å—è–≥–∞—î—Ç—å—Å—è –ø—Ä–∏ $$q = q_E$$:

–ê–Ω–∞–ª—ñ—Ç–∏—á–Ω–∏–π –ø–æ—à—É–∫ –º–∞–∫—Å–∏–º–∞–ª—å–Ω–æ–≥–æ –ø—Ä–∏–±—É—Ç–∫—É –ø–æ–ª—è–≥–∞—î —É —Ä–æ–∑–≤'—è–∑–∞–Ω–Ω—ñ –∑–∞–¥–∞—á—ñ

—Ç–æ–±—Ç–æ —É –∑–Ω–∞—Ö–æ–¥–∂–µ–Ω–Ω—ñ —Ä–æ–∑–≤'—è–∑–∫—É —Ä—ñ–≤–Ω—è–Ω–Ω—è

#### 4. –û–¥–Ω–æ–ø—Ä–æ–¥—É–∫—Ç–æ–≤–∞ —Ñ—ñ—Ä–º–∞ (3 –º–æ–¥–µ–ª—ñ –ø–æ–≤–µ–¥—ñ–Ω–∫–∏ —Ñ—ñ—Ä–º–∏).

–†–æ–∑–≥–ª—è–Ω–µ–º–æ –Ω–µ–æ–∫–ª–∞—Å–∏—á–Ω—É —Ç–µ–æ—Ä—ñ—é —É —è–∫—ñ–π —Ñ—ñ—Ä–º–∞ –º–∞—î –∑–∞–¥–∞–Ω—É –≤–∏—Ä–æ–±–Ω–∏—á—É —Ñ—É–Ω–∫—Ü—ñ—é $$q = F(x)$$, –∑–∞–¥–∞–Ω—É —Ü—ñ–Ω—É –≤–∏–ø—É—Å–∫—É, —Ç–∞ –∑–∞–¥–∞–Ω—ñ —Ü—ñ–Ω–∏ —Ñ–∞–∫—Ç–æ—Ä—ñ–≤ –≤–∏—Ä–æ–±–Ω–∏—Ü—Ç–≤–∞ (—Ä–µ—Å—É—Ä—ñ–≤), –ø—Ä–∏—á–æ–º—É –≤–≤–∞–∂–∞—î–º–æ —â–æ —Ñ—ñ—Ä–º–∞ –º–æ–∂–µ —Ä–µ–≥—É–ª—é–≤–∞—Ç–∏ —Å–≤—ñ–π –ø–æ–ø–∏—Ç –Ω–∞ —Ñ–∞–∫—Ç–æ—Ä–∏ –≤–∏—Ä–æ–±–Ω–∏—Ü—Ç–≤–∞.

–ú–æ–∂–Ω–∞ –ø–æ—Å—Ç–∞–≤–∏—Ç–∏ –∫—ñ–ª—å–∫–∞ –∑–∞–¥–∞—á, –∞ —Å–∞–º–µ:

1. –ú–∞–∫—Å–∏–º—ñ–∑–∞—Ü—ñ—ó –ø—Ä–∏–±—É—Ç–∫—É:

#### 5. –ú–æ–Ω–æ–ø–æ–ª—ñ—è, –º–æ–Ω–æ–ø—Å–æ–Ω—ñ—è.

–ß–∏—Å—Ç–∏–π –º–æ–Ω–æ–ø–æ–ª—ñ—Å—Ç –º–∞—î –∞–±—Å–æ–ª—é—Ç–Ω—É —Ä–∏–Ω–∫–æ–≤—É –≤–ª–∞–¥—É, –π–æ–≥–æ –∑–¥–∞—Ç–Ω—ñ—Å—Ç—å –≤–ø–ª–∏–≤–∞—Ç–∏ –Ω–∞ —Ü—ñ–Ω—É –æ–±–º–µ–∂—É—î—Ç—å—Å—è –ª–∏—à–µ –ø–æ–ø–∏—Ç–æ–º —Å–ø–æ–∂–∏–≤–∞—á—ñ–≤, —Ç–æ–º—É —Ñ—ñ—Ä–º–∞-–º–æ–Ω–æ–ø–æ–ª—ñ—Å—Ç —î _—Ü—ñ–Ω–æ—É—Ç–≤–æ—Ä—é–≤–∞—á–µ–º_, –Ω–∞ –≤—ñ–¥–º—ñ–Ω—É –≤—ñ–¥ –∫–æ–Ω–∫—É—Ä–µ–Ω—Ç–Ω–æ—ó —Ñ—ñ—Ä–º–∏.

–ö–æ–Ω–∫—É—Ä–µ–Ω—Ç–Ω–∏–π –º–µ—Ö–∞–Ω—ñ–∑–º –Ω–µ —Å–ø—Ä–∞—Ü—å–æ–≤—É—î –Ω–∞ –º–æ–Ω–æ–ø–æ–ª—ñ—Å—Ç–∏—á–Ω–∏–º—É —Ä–∏–Ω–∫—É —á–µ—Ä–µ–∑ –Ω–∞—è–≤–Ω—ñ—Å—Ç—å –≤—Ö—ñ–¥–Ω–∏—Ö –±–∞—Ä'—î—Ä—ñ–≤:

–§—ñ—Ä–º–∞-–º–æ–Ω–æ–ø–æ–ª—ñ—Å—Ç –º–∞—î –∑–º–æ–≥—É –≤–ø–ª–∏–≤–∞—Ç–∏ –Ω–∞ —Ü—ñ–Ω—É –ø—Ä–æ–¥—É–∫—Ü—ñ—ó —à–ª—è—Ö–æ–º –≤–∞—Ä—ñ—é–≤–∞–Ω–Ω—è –æ–±—Å—è–≥—ñ–≤ –≤–∏–ø—É—Å–∫—É —Å–≤–æ—î—ó –ø—Ä–æ–¥—É–∫—Ü—ñ—ó. –¢–∞–∫–∏–º —á–∏–Ω–æ–º —Ü—ñ–Ω–∞ –Ω–∞ –ø—Ä–æ–¥—É–∫—Ü—ñ—é $$p = p(q)$$, $$\mathrm{d} p / \mathrm{d} q < 0$$.

_–ú–æ–Ω–æ–ø—Å–æ–Ω—ñ—è_ &mdash; –º–æ–∂–ª–∏–≤—ñ—Å—Ç—å —á–∏–Ω–∏—Ç–∏ –≤–ø–ª–∏–≤ –Ω–∞ —Ü—ñ–Ω–∏ —Ñ–∞–∫—Ç–æ—Ä—ñ–≤ –≤–∏—Ä–æ–±–Ω–∏—Ü—Ç–≤–∞ (–º–æ–Ω–æ–ø–æ–ª—å–Ω–µ –≤–æ–ª–æ–¥—ñ–Ω–Ω—è –≤—Å—ñ–º –æ–±—Å—è–≥–æ–º –ø—Ä–æ–ø–æ–∑–∏—Ü—ñ—ó –æ–∫—Ä–µ–º–æ–≥–æ —Ä–µ—Å—É—Ä—Å—É).

–í–∏—Ä–æ–±–Ω–∏–∫ –º–æ–∂–µ –≤–ø–ª–∏–≤–∞—Ç–∏ –Ω–∞ —Ü—ñ–Ω—É —Ñ–∞–∫—Ç–æ—Ä—ñ–≤ –≤–∏—Ä–æ–±–Ω–∏—Ü—Ç–≤–∞, –∫–æ—Ä–∏—Å—Ç—É—é—á–∏—Å—å —Ç–∏–º, —â–æ –≤—ñ–Ω –∑–¥—ñ–π—Å–Ω—é—î –∑–∞–∫—É–ø—ñ–≤–ª—é —Ñ–∞–∫—Ç–æ—Ä—ñ–≤ –≤–∏—Ä–æ–±–Ω–∏—Ü—Ç–≤–∞ —É –∑–Ω–∞—á–Ω–∏—Ö —Ä–æ–∑–º—ñ—Ä–∞—Ö. 

–¢–∞–∫ —á–∏ —ñ–Ω–∞–∫—à–µ, –≤–≤–æ–¥–∏—Ç—å—Å—è $$w = w(x)$$ —è–∫–∞ —Ö–∞—Ä–∞–∫—Ç–µ—Ä–∏–∑—É—î –ø–ª–∞—Ç—É —Ñ—ñ—Ä–º–∏ –∑–∞ —Ä–µ—Å—É—Ä—Å–∏ –ø—Ä–∏ —Ä—ñ–∑–Ω–∏—Ö —Ä—ñ–≤–Ω—è—Ö –ø–æ–ø–∏—Ç—É –Ω–∞ –Ω–∏—Ö. –¢–æ–¥—ñ $$\mathrm{d} w / \mathrm{d} x > 0$$, $$\text{TC} = \langle w(x), x\rangle$$, —ñ –Ω–∞—Ä–µ—à—Ç—ñ

–Ø–∫ —ñ —Ä–∞–Ω—ñ—à–µ, –º–µ—Ç–∞ —Ñ—ñ—Ä–º–∏-–º–æ–Ω–æ–ø—Å–æ–Ω—ñ—Å—Ç–∞ –ø–æ–ª—è–≥–∞—î –≤ –º–∞–∫—Å–∏–º—ñ–∑–∞—Ü—ñ—ó –ø—Ä–∏–±—É—Ç–∫—É.

#### 6. –î—É–æ–ø–æ–ª—ñ—è –ö—É—Ä–Ω–æ, –®—Ç–∞–∫–∫–µ–ª—å–±–µ—Ä–≥–∞. –ì–∞–¥–∞–Ω—ñ –≤–∞—Ä—ñ–∞—Ü—ñ—ó.

_–û–ª—ñ–≥–æ–ø–æ–ª—ñ—è_ &mdash; —Ä–∏–Ω–∫–æ–≤–∞ —Å—Ç—Ä—É–∫—Ç—É—Ä–∞, –∑–∞ —è–∫–æ—ó –Ω–∞ —Ä–∏–Ω–∫—É –ø—Ä–æ–¥—É–∫—Ü—ñ—ó –ø—Ä–æ–ø–æ–∑–∏—Ü—ñ—è –Ω–µ–±–∞–≥–∞—Ç—å–æ—Ö —Ñ—ñ—Ä–º –∑–∞–ø–æ–≤–Ω—é—î –≤–µ—Å—å —Ä–∏–Ω–æ–∫ —ñ –¥–µ–∫—ñ–ª—å–∫–∞ –∑ —Ü–∏—Ö —Ñ—ñ—Ä–º –∑–∞–π–º–∞—é—Ç—å –∑–Ω–∞—á–Ω—ñ —á–∞—Å—Ç–∫–∏ —Ä–∏–Ω–∫—É.

_–û–ª—ñ–≥–æ–ø—Å–æ–Ω—ñ—è_ &mdash; —Ä–∏–Ω–∫–æ–≤–∞ —Å—Ç—Ä—É–∫—Ç—É—Ä–∞, –∑–∞ —è–∫–æ—ó –Ω–∞ —Ä–∏–Ω–∫—É —Ä–µ—Å—É—Ä—Å—ñ–≤ –ø–æ–ø–∏—Ç –Ω–∞ –ø–µ–≤–Ω—ñ —Ä–µ—Å—É—Ä—á–∏ —Ä–æ–∑–ø–æ–¥—ñ–ª–µ–Ω–∏–π —Å–µ—Ä–µ–¥ –Ω–µ–±–∞–≥–∞—Ç—å–æ—ó —Ñ—ñ—Ä–º, –¥–µ—è–∫—ñ –∑ —è–∫–∏—Ö –º–∞—é—Ç—å –∑–Ω–∞—á–Ω—ñ —á–∞—Å—Ç–∫–∏ –ø–æ–ø–∏—Ç—É.

#### 7. –ë–∞–≥–∞—Ç–æ–ø—Ä–æ–¥—É–∫—Ç–æ–≤–µ –≤–∏—Ä–æ–±–Ω–∏—Ü—Ç–≤–æ (2 –º–æ–¥–µ–ª—ñ –ø–æ–≤–µ–¥—ñ–Ω–∫–∏ —Ñ—ñ—Ä–º–∏).

$$

$$

—Ç–æ–±—Ç–æ –º–∞—î–º–æ $$n$$ –≤–∏–¥—ñ–≤ –ø—Ä–æ–¥—É–∫—Ü—ñ—ó, $$m$$ –≤–∏–¥—ñ–≤ —Ä–µ—Å—É—Ä—Å—ñ–≤, —ñ —Å—É–º–∞—Ä–Ω–µ —Å–ø–æ–∂–∏–≤–∞–Ω–Ω—è —Ä–µ—Å—É—Ä—Å—ñ–≤ –≤–∏–∑–Ω–∞—á–∞—î—Ç—å—Å—è —è–∫

$$

$$

–∞ –≤–µ–∫—Ç–æ—Ä —Ü—ñ–Ω –ø—Ä–æ–¥—É–∫—Ü—ñ—ó —Ñ—ñ–∫—Å–æ–≤–∞–Ω–∏–π, $$p = (p_i)_{i = 1}^n$$.

–¢–æ–¥—ñ

$$
\pi(x) = \langle p, F \rangle - \langle w, x \rangle = \displaystyle\sum\limits_{i = 1}^n p_i F_i(x^i) - \displaystyle\sum\limits_{j = 1}^m w_j \left( \displaystyle\sum\limits_{k=1}^n x_j^k \right).
$$
